def simulate_stock(start, rate, vol,days):
    daily_yeilds = np.random.normal((rate/252),vol/(252**0.5), size=days)
    cumulative_yields = np.cumsum(daily_yeilds)
    daily_multipliers= np.exp(cumulative_yields)
    daily_price = np.round(daily_multipliers * start,2)
    daily_price_array =  np.concatenate((start,daily_price),axis=None)
    return(daily_price_array)

def find_yield(run):
    annual_yield = round((math.log(run[60]/run[0]))*252/(len(run)-1),4)
    return(annual_yield)

Part D: Monte Carlo Simulation¶

Running the same simulation 10k times to see how the data spreads out with high number of runs¶

In [11]:
def monte_carlo(start, rate, vol, days, num_runs):
    final_prices = []
    annual_yields = []
    for run in range(0,num_runs):
        run_array = simulate_stock(start, rate, vol, days)
        final_yield = find_yield(run_array)
        annual_yields.append(final_yield)
        final_prices.append(run_array[days])
    return[np.array(final_prices),np.array(annual_yields)]
