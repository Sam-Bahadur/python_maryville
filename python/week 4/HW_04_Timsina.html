def  classification_report(true_y,pred_y):
    classes = np.unique(true_y)
    accuracy = find_accuracy(true_y,pred_y)
    TP = np.sum(pred_y == classes[1])
    FP = np.sum(true_y == classes[0])
    TN = np.sum(pred_y==classes[1])
    FN = np.sum(true_y==classes[0])
    positive_precision = TP/(TP+FP)
    positive_recall = TP/(TP+FN)
    negative_precision = TN/(TN+FN)
    negative_recall = TN/(TN+FP)
    print("Positive Class :\t ", classes[1])
    print("Negetive Class :\t ", classes[0])
    print("")
    print("Accuracy:\t\t ",accuracy)
    print("Positive Precision :\t ", positive_precision)
    print("Positive Recall :\t ", positive_recall)
    print("Negetive Precision :\t ", negative_precision)
    print("Negative Recall :\t ", negative_recall)
